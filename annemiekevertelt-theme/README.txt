<?php
/**
 * Annemieke Vertelt - WordPress Theme
 * 
 * Installatie instructies:
 * 
 * 1. WordPress Installatie:
 *    - Ga naar https://annemiekevertelt.nl/wp-admin/install.php
 *    - Kies taal: Nederlands
 *    - Vul in:
 *      * Site titel: Annemieke vertelt
 *      * Gebruikersnaam: (kies een veilige naam, NIET 'admin')
 *      * Wachtwoord: (genereer een sterk wachtwoord)
 *      * Email: jouw@email.nl
 *    - Klik "WordPress installeren"
 * 
 * 2. Theme Upload:
 *    - Log in op wp-admin
 *    - Ga naar "Weergave" > "Thema's" > "Nieuw toevoegen" > "Theme uploaden"
 *    - Selecteer annemiekevertelt-theme.zip
 *    - Activeer het theme
 * 
 * 3. Permalinks instellen:
 *    - Ga naar "Instellingen" > "Permalinks"
 *    - Kies "Berichtnaam"
 *    - Sla op
 * 
 * 4. Homepage instellen:
 *    - Ga naar "Pagina's" > "Nieuwe pagina"
 *    * Titel: Home
 *    * Template: Homepage (rechts in sidebar)
 *    - Publiceer
 *    - Ga naar "Instellingen" > "Lezen"
 *    * Kies "Een statische pagina" > Home
 *    - Sla op
 * 
 * 5. Menu's maken:
 *    - Ga naar "Weergave" > "Menu's"
 *    * Maak menu "Hoofdmenu"
 *    * Voeg toe: Home, Columns, Kinderverhalen, Audio, Wie ben ik?
 *    * Locatie: Hoofdmenu
 * 
 * 6. Content toevoegen:
 *    - Columns: Ga naar "Columns" > "Nieuwe column"
 *    - Kinderverhalen: Ga naar "Kinderverhalen" > "Nieuw verhaal"
 *    - Audio: Ga naar "Audio" > "Nieuw audio verhaal"
 * 
 * Aangepaste velden:
 * - Audio berichten: gebruik een audioblock of embed SoundCloud/Spotify
 * 
 * Shortcodes:
 * - [mocht_je_me_gemist_hebben] - Toont recente berichten
 */
